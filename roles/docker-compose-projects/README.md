Service definition examples

```yaml
docker_compose_services:
  - name: test-service
    compose_files:
      - filename: docker-compose-1.yml
        # When template already contains Ansible managed warning
        ansible_managed_header: false
        # 1. Template in inventory templates subdir
        content: "{{ lookup('ansible.builtin.template', inventory_dir + '/templates/compose_template_test.yml.j2') }}"

      - filename: docker-compose-2.yml
        # 2. Template in collection templates (relative to ansible-playbooks/docker_compose_projects.yml)
        content: "{{ lookup('ansible.builtin.template', 'compose_template_test.yml.j2') }}"

      - filename: docker-compose-3.yml
        # 3. Regular file on the controller (use expanduser filter to expand ~)
        content: "{{ lookup('ansible.builtin.file', ('~/projects/docker-configs/seafile/docker-compose.yml' | expanduser)) }}"

      - filename: docker-compose-4.yml
        # 4. Content from HTTP(S) URL
        content: 
          {{ lookup('ansible.builtin.url', 'https://github.com/cheretbe/docker-configs/raw/refs/heads/main/seafile/docker-compose.https.yml', split_lines=False) }}

      - filename: docker-compose-5.yml
        ansible_managed_header: false
        # 5. Directly setting content
        content: |
          ---

          # (custom) Generated by Ansible at {{ lookup('pipe', 'hostname -f') }}
          # Any changes made here will be overwritten

          {{ lookup('ansible.builtin.url', 'https://github.com/cheretbe/docker-configs/raw/refs/heads/main/seafile/docker-compose.yml', split_lines=False) }}

        # 6. Setting content from a file on a target host is not possible because filters run on the controller.
        #    In the unlikely event something like this will be necessary, possible solution is to use
        #    ansible.builtin.slurp or ansible.builtin.copy (with remote_src: true) and explicitly call
        #    community.docker.docker_compose_v2
```
